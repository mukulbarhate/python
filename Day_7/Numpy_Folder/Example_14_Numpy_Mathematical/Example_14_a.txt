# axis=0  column-wise sum
# axis=1  row-wise sum

import numpy as k
arr=k.array([[1,2,3,4],
             [2,3,4,5],
             [7,8,9,10]])
arr1=k.sum(arr,axis=1)   # row-wise total
# 1+2+3+4   2+3+4+5   7+8+9+10
print(arr1)
arr2=k.sum(arr,axis=0)  # column-wise total
# 1+2+7     2+3+8     3+4+9     4+5+10
print(arr2)
