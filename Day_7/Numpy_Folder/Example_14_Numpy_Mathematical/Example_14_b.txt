import numpy as np

fourth=np.array([[10,20,30],[40,50,60]])
fifth=np.array([[70,80,90],[100,110,120]])
sixth=np.sum((fourth,fifth),axis=0)  
print(sixth)
""" [[ 80 100 120] 
     [140 160 180]] """
print("\n")
eighth=np.sum((fourth,fifth),axis=1)
print(eighth)
""" [[ 50  70  90] k
 [170 190 210]] """
print()
ninth=np.sum((fourth,fifth),axis=2)
print(ninth)
""" [[ 60 150]     
 [240 330]] """
