class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class Queue:
    def __init__(self):
        self.head = None
        self.last=None
    def add(self,data):
       ref=Node(data)
       if self.head==None:
           self.head=ref
           self.last=ref
       else:
           self.last.next=ref
           self.last=ref
    def display(self):
       if self.head is None:
          print("Queue is empty")
          return
       else:
          cur=self.head
          while(cur is not None):
            print(cur.data)
            cur=cur.next

    def remove(self):
        if self.head is None:
            print("queue is empty")
            return
        cur=self.head
        self.head=cur.next
        cur=None

myqueue=Queue()
myqueue.add(10)
myqueue.add(20)
myqueue.add(30)
myqueue.add(40)
print("let's print entire queue")
myqueue.display()
print("let's remove the first element")
myqueue.remove()
print("let's print entire queue after remove operation")
myqueue.display()
